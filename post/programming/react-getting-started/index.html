React - Environment setup

Basic configuration for React
Share

React + webpack + babel/ES6 + SASS + normalize

Global libraries

  
    npm install -g babel babel-cli

In case you are using Windows 10

  
    npm install --global --production windows-build-tools
    npm config set msvs_version 2015 --global

Project setup

Init npm inside the project folder and setup the start script in the package.json

  
    npm init

package.json start script:

  
    "start": "webpack-dev-server --hot"

Dependencies

  
    npm install --save-dev babel-core babel-loader babel-preset-react babel-preset-es2015 webpack-dev-server sass-loader css-loader style-loader node-sass
    npm install --save webpack react react-dom normalize-scss

The following lines are to set up the linter

  
    npm install --save-dev eslint eslint-config-airbnb eslint-plugin-import eslint-plugin-jsx-a11y eslint-plugin-react

Files setup

webpack.config.js

  
    const config = {
      entry: './main.js',
      output: {
        path: '/',
        filename: 'index.js',
      },
      devServer: {
        inline: true,
        port: 7777,
      },
      module: {
        loaders: [ //Cambia a rules'???
          {
            test: /\.jsx?$/,
            exclude: /node_modules/,
            loader: 'babel-loader',
            query: {
              presets: ['es2015', 'react'],
            },
          },
        ],
      },
    };

    module.exports = config;

.eslintrc.js

  
    module.exports = {
      "extends": "airbnb",
      "env": {
        "browser": true,
        "node": true
      },
      "plugins": [
        "react",
        "jsx-a11y",
        "import"
      ],
      "rules": {
        "react/jsx-filename-extension": [1, { "extensions": [".js", ".jsx"] }],
        "linebreak-style": ["error", "windows"],
      }
    };
